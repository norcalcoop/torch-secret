**Regenerate function (auto-called on settings change and on Regenerate button click):**
```typescript
function regenerate(): void {
  try {
    const result = generatePassword({
      tier: activeTier,
      uppercase: uppercaseCheckbox.checked,
      numbers: numbersCheckbox.checked,
      symbols: symbolsCheckbox.checked,
      easyToSay: easyToSayCheckbox.checked,
      easyToRead: easyToReadCheckbox.checked,
      omitSimilar: omitSimilarCheckbox.checked,
    });
    previewPassword = result.password;
    previewField.textContent = previewPassword;
    // Update entropy line: "High · ~105 bits · ~centuries at 10B guesses/sec"
    const tierLabel = activeTier.charAt(0).toUpperCase() + activeTier.slice(1);
    entropyLine.textContent = `${tierLabel} · ~${Math.round(result.entropyBits)} bits · ${result.bruteForceEstimate}`;
    genError.classList.add('hidden');
  } catch {
    genError.classList.remove('hidden');
    previewField.textContent = '';
    previewPassword = '';
  }
}
```
