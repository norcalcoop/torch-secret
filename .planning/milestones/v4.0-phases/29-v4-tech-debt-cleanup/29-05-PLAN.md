showRateLimitUpsell() is defined at line 178 of client/src/pages/create.ts. It is NOT exported â€” it is a module-private function. The test must access it indirectly.

**Indirect access strategy:** The function is called by the submit handler when an ApiError with status 429 is caught. However, for a unit test the cleanest approach is to test the observable DOM output by calling renderCreatePage() and simulating a 429 ApiError through the form submit event â€” OR, more directly, by temporarily exporting the function for test access. Because the project uses ESM and Vitest runs in the same module scope as the source, the simplest and most robust approach is to test via module augmentation: mock the `../api/client.js` module so `createSecret` throws an `ApiError` with `status: 429` and a specific `rateLimitReset` value, then submit the form, and assert the DOM.

**How ApiError works (from client/src/api/client.ts):** Import `ApiError` from `../api/client.js`. It has `status: number` and `rateLimitReset?: number` properties. The `rateLimitReset` field is the delta-seconds value from the `RateLimit-Reset` header.

**What showRateLimitUpsell() does (lines 178-222 of create.ts):**
- Clears the errorArea element and replaces its className with neutral styling
- Sets role="alert" on the container
- Appends a `<p>` with "You've reached the free limit for anonymous sharing."
- If `resetTimestamp && resetTimestamp > 0`: computes `minutesUntilReset = Math.ceil(resetTimestamp / 60)` and appends `<p>` with text "Limit resets in N minute(s)." (singular if N === 1) or "Limit resets soon." if minutesUntilReset is 0
- Appends a benefit line and a CTA `<a>` to /register

// @vitest-environment happy-dom

/**
 * Unit tests for showRateLimitUpsell() in client/src/pages/create.ts.
 *
 * showRateLimitUpsell() is module-private. These tests exercise it by
 * rendering the create page, mocking createSecret() to throw a 429 ApiError,
 * and submitting the form â€” then asserting the DOM output.
 *
 * This provides the alternative coverage referenced in the skip comment of
 * e2e/specs/rate-limits.spec.ts (lines 7-9), which is permanently skipped
 * because playwright.config.ts hardcodes E2E_TEST=true.
 */

import { describe, it, expect, beforeEach, afterEach, vi } from 'vitest';

// Mock the crypto module (Web Crypto encrypt is not available in happy-dom)
vi.mock('../crypto/index.js', () => ({
  encrypt: vi.fn().mockResolvedValue({
    payload: { ciphertext: 'dGVzdA==' },
    keyBase64Url: 'testkey',
  }),
  generatePassphrase: vi.fn().mockReturnValue('word-word-word-word'),
  generatePassword: vi.fn().mockReturnValue({
    password: 'TestPass1!',
    entropyBits: 52,
    bruteForceEstimate: 'centuries',
  }),
}));

// Mock PostHog analytics (no network calls in unit tests)
vi.mock('../analytics/posthog.js', () => ({
  captureSecretCreated: vi.fn(),
  captureConversionPromptShown: vi.fn(),
  captureConversionPromptClicked: vi.fn(),
}));

// Mock the auth client (prevents network call; anonymous user = no session)
vi.mock('../api/auth-client.js', () => ({
  authClient: {
    getSession: vi.fn().mockResolvedValue({ data: null }),
  },
}));

// Mock router navigate (prevents navigation side-effects)
vi.mock('../router.js', () => ({
  navigate: vi.fn(),
}));

// ApiError is the real class â€” import before mocking createSecret
import { ApiError } from '../api/client.js';

// Mock createSecret to simulate 429 responses with configurable rateLimitReset
const mockCreateSecret = vi.fn();
vi.mock('../api/client.js', async (importOriginal) => {
  const original = await importOriginal<typeof import('../api/client.js')>();
  return {
    ...original,
    createSecret: (...args: unknown[]) => mockCreateSecret(...args),
  };
});

let container: HTMLDivElement;

beforeEach(() => {
  container = document.createElement('div');
  container.id = 'app';
  document.body.appendChild(container);
  vi.clearAllMocks();
});

afterEach(() => {
  document.body.removeChild(container);
});

/**
 * Helper: render the create page, fill in the textarea, and submit the form.
 * createSecret is expected to be mocked to throw before this is called.
 */
async function submitCreateForm(): Promise<void> {
  const { renderCreatePage } = await import('../pages/create.js');
  renderCreatePage(container);

  const textarea = container.querySelector<HTMLTextAreaElement>('#secret-text');
  expect(textarea).not.toBeNull();
  textarea!.value = 'test secret content';
  // Trigger the input event so the form considers the field non-empty
  textarea!.dispatchEvent(new Event('input'));

  const form = container.querySelector<HTMLFormElement>('form');
  expect(form).not.toBeNull();
  form!.dispatchEvent(new Event('submit', { bubbles: true, cancelable: true }));

  // Wait for the async submit handler to resolve
  await new Promise((resolve) => setTimeout(resolve, 50));
}

describe('showRateLimitUpsell() â€” countdown rendering', () => {
  it('renders "Limit resets in 60 minutes." when resetTimestamp is 3600', async () => {
    // 3600 seconds â†’ Math.ceil(3600 / 60) = 60 minutes
    mockCreateSecret.mockRejectedValue(
      Object.assign(new ApiError('Rate limit exceeded', 429), { rateLimitReset: 3600 }),
    );

    await submitCreateForm();

    const alertEl = container.querySelector('[role="alert"]');
    expect(alertEl).not.toBeNull();
    expect(alertEl!.textContent).toContain("You've reached the free limit for anonymous sharing.");

    // 60 minutes â€” plural
    const countdownEl = Array.from(alertEl!.querySelectorAll('p')).find((p) =>

    );
    expect(countdownEl).not.toBeNull();
  });

  it('renders "Limit resets in 1 minute." (singular) when resetTimestamp is 45', async () => {
    // 45 seconds â†’ Math.ceil(45 / 60) = 1 minute â†’ singular
    mockCreateSecret.mockRejectedValue(
      Object.assign(new ApiError('Rate limit exceeded', 429), { rateLimitReset: 45 }),
    );

    await submitCreateForm();

    const alertEl = container.querySelector('[role="alert"]');
    expect(alertEl).not.toBeNull();

    const countdownEl = Array.from(alertEl!.querySelectorAll('p')).find((p) =>
      p.textContent?.includes('Limit resets in 1 minute.'),
    );
    expect(countdownEl).not.toBeNull();
  });

  it('renders "Limit resets in 30 minutes." when resetTimestamp is 1800', async () => {
    // 1800 seconds â†’ Math.ceil(1800 / 60) = 30 minutes
    mockCreateSecret.mockRejectedValue(
      Object.assign(new ApiError('Rate limit exceeded', 429), { rateLimitReset: 1800 }),
    );

    await submitCreateForm();

    const alertEl = container.querySelector('[role="alert"]');
    expect(alertEl).not.toBeNull();

    const countdownEl = Array.from(alertEl!.querySelectorAll('p')).find((p) =>
      p.textContent?.includes('Limit resets in 30 minutes.'),
    );
    expect(countdownEl).not.toBeNull();
  });

  it('renders no countdown paragraph when resetTimestamp is undefined', async () => {
    // No rateLimitReset â€” the "if (resetTimestamp && resetTimestamp > 0)" branch is skipped
    mockCreateSecret.mockRejectedValue(
      Object.assign(new ApiError('Rate limit exceeded', 429), { rateLimitReset: undefined }),
    );

    await submitCreateForm();

    const alertEl = container.querySelector('[role="alert"]');
    expect(alertEl).not.toBeNull();

      p.textContent?.match(/Limit resets/),

  });

  it('renders "Sign up" CTA link pointing to /register', async () => {
    mockCreateSecret.mockRejectedValue(
      Object.assign(new ApiError('Rate limit exceeded', 429), { rateLimitReset: 3600 }),

    await submitCreateForm();

    const alertEl = container.querySelector('[role="alert"]');
    expect(alertEl).not.toBeNull();

    const ctaLink = alertEl!.querySelector<HTMLAnchorElement>('a[href="/register"]');
    expect(ctaLink).not.toBeNull();

  });
});

**Important details:**
- The file header `// @vitest-environment happy-dom` is required â€” the client Vitest project also sets `environment: 'happy-dom'` but the in-file directive ensures IDE tooling and isolated runs use the correct environment.
- `vi.mock('../api/client.js', async (importOriginal) => {...})` uses the factory form to preserve the real `ApiError` class while replacing only `createSecret`. This is required because `ApiError` must be the real class for `instanceof` checks in create.ts (line 1165: `err instanceof ApiError && err.status === 429`).
- `Object.assign(new ApiError(...), { rateLimitReset: N })` attaches the `rateLimitReset` property to the real ApiError instance. Check `client/src/api/client.ts` to confirm the actual ApiError constructor signature and `rateLimitReset` field name â€” adjust if needed.
- The 50ms `setTimeout` flush after form submit is sufficient for the `void (async () => {...})()` IIFE in the submit handler to resolve.
- If `ApiError` constructor does not accept `rateLimitReset` in its constructor, use `Object.assign` to add the property after construction (as shown above).
- Do NOT import `showRateLimitUpsell` directly â€” it is not exported. All assertions are against the rendered DOM.
