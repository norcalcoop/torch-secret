    2. Create server/src/auth.ts. This is the single source of truth for Better Auth configuration.
       Key constraints from research:
       - Use drizzleAdapter with usePlural: true AND explicit schema mapping (user: schema.users)
         because our tables use plural names but Better Auth's adapter uses singular internally.
       - requireEmailVerification: true only when NODE_ENV !== 'test' to allow integration tests
         to bypass the email gate (test users cannot receive real emails).
       - Google and GitHub socialProviders: only include in config when env vars are defined.
         Use a pattern like: ...(env.GOOGLE_CLIENT_ID ? { google: {...} } : {})
       - sendResetPassword: use void to avoid awaiting (prevents timing attacks leaking email existence).
         Construct the reset URL manually as BETTER_AUTH_URL + /reset-password?token= because of
         the known Better Auth inconsistency in reset URL format (research Pitfall 7). Pass the raw
         url arg from the callback as-is — let it be whatever Better Auth generates — but note that
         the frontend reset-password page (Plan 05) must handle both /reset-password?token= and
         /api/auth/reset-password?token= formats by extracting the token query parameter.
       - Never call auth.migrate() — Phase 21 already applied migrations.
       - Export: auth, AuthSession type, AuthUser type, AuthSessionData type.

       Full implementation:

       import { betterAuth } from 'better-auth';
       import { drizzleAdapter } from 'better-auth/adapters/drizzle';
       import { db } from './db/connection.js';
       import * as schema from './db/schema.js';
       import { resend } from './services/email.js';
       import { env } from './config/env.js';

       export const auth = betterAuth({
         baseURL: env.BETTER_AUTH_URL,
         secret: env.BETTER_AUTH_SECRET,

         database: drizzleAdapter(db, {
           provider: 'pg',
           schema: {
             ...schema,
             user: schema.users,
           },
           usePlural: true,
         }),

         emailAndPassword: {
           enabled: true,
           requireEmailVerification: env.NODE_ENV !== 'test',
           minPasswordLength: 8,
           sendResetPassword: async ({ user, url }) => {
             void resend.emails.send({
               from: env.RESEND_FROM_EMAIL,
               to: user.email,
               subject: 'Reset your SecureShare password',
               text: `Click to reset your password: ${url}\n\nIf you did not request this, you can ignore this email.`,
             });
           },
         },

         emailVerification: {
           sendVerificationEmail: async ({ user, url }) => {
             void resend.emails.send({
               from: env.RESEND_FROM_EMAIL,
               to: user.email,
               subject: 'Verify your SecureShare email',
               text: `Click to verify your email: ${url}\n\nIf you did not create an account, you can ignore this email.`,
             });
           },
           sendOnSignIn: true,
         },

         socialProviders: {
           ...(env.GOOGLE_CLIENT_ID && env.GOOGLE_CLIENT_SECRET
             ? {
                 google: {
                   clientId: env.GOOGLE_CLIENT_ID,
                   clientSecret: env.GOOGLE_CLIENT_SECRET,
                 },
               }
             : {}),
           ...(env.GITHUB_CLIENT_ID && env.GITHUB_CLIENT_SECRET
             ? {
                 github: {
                   clientId: env.GITHUB_CLIENT_ID,
                   clientSecret: env.GITHUB_CLIENT_SECRET,
                 },
               }
             : {}),
         },

         trustedOrigins: [env.BETTER_AUTH_URL],
       });

       export type AuthSession = typeof auth.$Infer.Session;
       export type AuthUser = typeof auth.$Infer.Session.user;
       export type AuthSessionData = typeof auth.$Infer.Session.session;

       ZERO-KNOWLEDGE INVARIANT REMINDER: The session object contains userId.
       No auth hook, middleware, or route may combine res.locals.user.id with a secretId
       in the same log line, DB record, or API response. See .planning/INVARIANTS.md.
  </action>
