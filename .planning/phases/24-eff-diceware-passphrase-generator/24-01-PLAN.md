    ```typescript
    /**
     * EFF Diceware Passphrase Generator
     *
     * Uses the EFF Large Wordlist (7,776 words = 6^5 combinations).
     * Source: https://www.eff.org/files/2016/07/18/eff_large_wordlist.txt
     * Published 2016. Stable — no updates since release.
     *
     * Security: Only crypto.getRandomValues may be used for randomness.
     * No Math.random. No third-party libraries. Project invariant (CLAUDE.md).
     *
     * Entropy: 4 words = log2(7776^4) ≈ 51.7 bits
     *          6 words = log2(7776^6) ≈ 77.5 bits (v5.0 Pro tier: PRO-04)
     */

    const EFF_WORDS: string[] = [
      // All 7,776 words inserted here as a flat array literal
      // Type is string[], NOT readonly [...] — avoids TS compile-time slowdown
      'abacus', 'abdomen', /* ... */ 'zoom',
    ];

    const WORD_COUNT = 7776; // Must equal EFF_WORDS.length

    /**
     * Return a single unbiased random index in [0, WORD_COUNT) using rejection sampling.
     *
     * Rejection cutoff: 2**32 - (2**32 % WORD_COUNT) = 4294964736.
     * Values >= cutoff are rejected. Rejection probability ≈ 0.0000006 — effectively zero.
     *
     * Invariant: uses only crypto.getRandomValues (project rule: no Math.random).
     */
    function getUnbiasedIndex(): number {
      const cutoff = 2 ** 32 - (2 ** 32 % WORD_COUNT);
      const buf = new Uint32Array(1);
      while (true) {
        crypto.getRandomValues(buf);
        if (buf[0] < cutoff) {
          return buf[0] % WORD_COUNT;
        }
      }
    }

    /**
     * Generate an EFF Diceware passphrase.
     *
     * @param wordCount - Number of words (default 4 for Phase 24).
     *                    Pass 6 for PRO-04 (v5.0) — same function, different argument.
     * @returns Space-separated lowercase passphrase (e.g. "abacus zoning climb velvet")
     */
    export function generatePassphrase(wordCount = 4): string {
      const words: string[] = [];
      for (let i = 0; i < wordCount; i++) {
        words.push(EFF_WORDS[getUnbiasedIndex()] as string);
      }
      return words.join(' ');
    }
    ```

    Note: `EFF_WORDS[index]` returns `string | undefined` in TS strict mode, hence the `as string` cast — the index is always valid since getUnbiasedIndex() returns [0, 7775].
