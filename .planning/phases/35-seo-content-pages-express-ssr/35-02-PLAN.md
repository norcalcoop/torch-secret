**A. `server/src/routes/seo/use.ts`:**

```typescript
import { Router } from 'express';
import { renderLayout } from './templates/layout.js';
import { USE_CASE_PAGES, USE_CASE_HUB } from './templates/use-case-pages.js';
import { escHtml } from './templates/layout.js';

export const useRouter = Router();

// Hub page: /use/
useRouter.get('/', (req, res) => {
  const nonce = res.locals.cspNonce as string;
  const hub = USE_CASE_HUB;

  const cardsHtml = hub.cards
    .map(
      (card) => `
      <a href="/use/${escHtml(card.slug)}" class="block rounded-xl border border-border bg-surface p-6 hover:bg-surface-raised transition-colors no-underline">
        <h2 class="font-heading text-lg font-semibold text-text-primary mb-2">${escHtml(card.title)}</h2>
        <p class="text-sm text-text-secondary">${escHtml(card.description)}</p>
      </a>`,
    )
    .join('\n');

  const bodyHtml = `
    <div class="mb-10">
      <h1 class="font-heading text-3xl font-bold text-text-primary mb-3">Secure Credential Sharing Guides</h1>
      <p class="text-text-secondary text-lg leading-relaxed">
        Step-by-step guides for sharing API keys, database passwords, SSH keys, .env files, and other credentials securely.
        Each guide shows how to use one-time encrypted links that self-destruct after a single view.
      </p>
    </div>
    <div class="grid gap-4 sm:grid-cols-2 lg:grid-cols-3">
      ${cardsHtml}
    </div>
    <div class="mt-12 border-t border-border pt-8">
      <h2 class="font-heading text-lg font-semibold text-text-primary mb-4">Also compare Torch Secret with alternatives</h2>
      <div class="flex flex-wrap gap-3 text-sm">
        <a href="/vs/onetimesecret" class="text-accent hover:underline">vs. OneTimeSecret</a>
        <a href="/vs/pwpush" class="text-accent hover:underline">vs. Password Pusher</a>
        <a href="/vs/privnote" class="text-accent hover:underline">vs. Privnote</a>
        <a href="/alternatives/onetimesecret" class="text-accent hover:underline">OneTimeSecret alternatives</a>
      </div>
    </div>`;

  const html = renderLayout({
    ...hub.meta,
    bodyHtml,
    jsonLd: '',
    cspNonce: nonce,
  });
  res.setHeader('Content-Type', 'text/html; charset=utf-8');
  res.send(html);
});

// Individual use-case page: /use/:slug
useRouter.get('/:slug', (req, res) => {
  const page = USE_CASE_PAGES[req.params.slug];
  if (!page) {
    res.status(404).json({ error: 'not_found' });
    return;
  }
  const nonce = res.locals.cspNonce as string;

  const howToSchema = {
    '@context': 'https://schema.org',
    '@type': 'HowTo',
    name: page.h1,
    description: page.description,
    step: page.steps.map((s, i) => ({
      '@type': 'HowToStep',
      position: i + 1,
      name: s.name,
      text: s.text,
    })),
  };

  const faqSchema = {
    '@context': 'https://schema.org',
    '@type': 'FAQPage',
    mainEntity: page.faqItems.map(({ question, answer }) => ({
      '@type': 'Question',
      name: question,
      acceptedAnswer: { '@type': 'Answer', text: answer },
    })),
  };

  // Combine both schemas into a single JSON array for one <script> block
  const jsonLd = JSON.stringify([howToSchema, faqSchema]);

  const html = renderLayout({
    ...page.meta,
    bodyHtml: page.bodyHtml,
    jsonLd,
    cspNonce: nonce,
  });
  res.setHeader('Content-Type', 'text/html; charset=utf-8');
  res.send(html);
});
```
