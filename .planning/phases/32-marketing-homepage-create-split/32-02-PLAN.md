```typescript
import { Flame, KeyRound, StickyNote, ShieldCheck, Lock, Zap } from 'lucide';
import { createIcon } from '../components/icons.js';
import { navigate } from '../router.js';
import { showToast } from '../components/toast.js';

export function renderHomePage(container: HTMLElement): void {
  const wrapper = document.createElement('div');
  wrapper.className = 'space-y-16 pb-16 sm:pb-8';

  wrapper.appendChild(createHeroSection());
  wrapper.appendChild(createTrustStrip());   // zero-knowledge proof points
  wrapper.appendChild(createUseCasesSection());
  wrapper.appendChild(createEmailCaptureSection());

  container.appendChild(wrapper);
}
```
